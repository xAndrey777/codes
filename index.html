package kripton.dev.module.modules.combat;

import kripton.dev.friend.FriendManager;
import kripton.dev.module.Category;
import kripton.dev.module.Module;
import kripton.dev.module.ModuleManager;
import kripton.dev.module.settings.BooleanSetting;
import kripton.dev.module.settings.ModeSetting;
import kripton.dev.module.settings.NumberSetting;
import kripton.dev.module.settings.StringSetting;
import net.minecraft.world.InteractionHand;
import net.minecraft.world.entity.Entity;
import net.minecraft.world.entity.LivingEntity;
import net.minecraft.world.entity.monster.Monster;
import net.minecraft.world.entity.player.Player;
import net.minecraft.world.item.ItemStack;
import net.minecraft.world.phys.Vec3;

import java.util.Arrays;
import java.util.Comparator;
import java.util.stream.StreamSupport;

public class KillAura extends Module {

    private final NumberSetting range   = new NumberSetting(\"Range\",  \"Distancia maxima de ataque (ojo->hitbox)\", 4.5, 2.0, 8.0, 0.1);
    private final NumberSetting cps     = new NumberSetting(\"CPS\",    \"Ataques por segundo\", 12.0, 1.0, 20.0, 0.5);
    private final NumberSetting fov     = new NumberSetting(\"FOV\",    \"Angulo de vision (360 = libre)\", 360.0, 30.0, 360.0, 5.0);

    private final BooleanSetting players     = new BooleanSetting(\"Players\",   \"Ataca a otros jugadores\", true);
    private final BooleanSetting mobs        = new BooleanSetting(\"Mobs\",      \"Ataca a mobs hostiles\",  true);
    private final BooleanSetting ignoreInv   = new BooleanSetting(\"Ignorar Invisibles\", \"Ignora entidades invisibles\", false);
    private final BooleanSetting rotate      = new BooleanSetting(\"Rotacion\",  \"Rota hacia el target (silent)\", true);
    private final BooleanSetting useCritical = new BooleanSetting(\"Criticals\", \"Fuerza criticos si el modulo esta activo\", true);

    private final ModeSetting weapon = new ModeSetting(\"Weapon\", \"Arma requerida\",
            \"Cualquiera\", Arrays.asList(\"Cualquiera\",\"Espada\",\"Hacha\",\"Pico\",\"Pala\",\"Tridente\",\"Mano\",\"Custom\"));
    private final StringSetting custom = new StringSetting(\"Custom Item\", \"Substring del item (ej: mace, netherite_sword)\", \"sword\");

    private long lastAttackTime = 0L;

    public KillAura() {
        super(\"KillAura\", \"Ataca automaticamente a entidades cercanas\", Category.COMBAT);
        addSetting(range);
        addSetting(cps);
        addSetting(fov);
        addSetting(players);
        addSetting(mobs);
        addSetting(ignoreInv);
        addSetting(rotate);
        addSetting(useCritical);
        addSetting(weapon);
        addSetting(custom);
    }

    @Override protected void onEnable() { lastAttackTime = System.currentTimeMillis(); }

    @Override
    public void onUpdate() {
        if (mc.player == null || mc.level == null || mc.gameMode == null) return;
        if (!hasCorrectWeapon()) return;

        long now = System.currentTimeMillis();
        long delay = (long) (1000.0 / cps.getValue());
        if (now - lastAttackTime < delay) return;

        Iterable<Entity> entities = mc.level.entitiesForRendering();
        Vec3 eye = mc.player.getEyePosition();
        double maxR = range.getValue();

        Entity target = StreamSupport.stream(entities.spliterator(), false)
                .filter(e -> isValidTarget(e, eye, maxR))
                .min(Comparator.comparingDouble(e -> e.getBoundingBox().distanceToSqr(eye)))
                .orElse(null);

        if (target == null) return;

        if (rotate.getValue()) {
            float[] rot = computeRotation(eye, target);
            // Rotacion silenciosa: solo afecta packets de movimiento
            mc.player.setYRot(rot[0]);
            mc.player.setXRot(rot[1]);
            mc.player.yRotO = rot[0];
            mc.player.xRotO = rot[1];
        }

        if (useCritical.getValue()) {
            Module crit = ModuleManager.getInstance().getModule(Criticals.class);
            if (crit != null && crit.isEnabled()) ((Criticals) crit).doCritical();
        }

        mc.gameMode.attack(mc.player, target);
        mc.player.swing(InteractionHand.MAIN_HAND);
        mc.player.resetAttackStrengthTicker();
        lastAttackTime = now;
    }

    private boolean isValidTarget(Entity entity, Vec3 eye, double maxR) {
        if (entity == mc.player) return false;
        if (!(entity instanceof LivingEntity living)) return false;
        if (living.isDeadOrDying() || living.getHealth() <= 0f) return false;
        if (ignoreInv.getValue() && living.isInvisible()) return false;

        // Distancia OJO -> HITBOX (es lo que valida el server)
        double distSqr = living.getBoundingBox().distanceToSqr(eye);
        if (distSqr > maxR * maxR) return false;

        // FOV check
        if (fov.getValue() < 360.0) {
            float ang = angleTo(eye, living);
            if (Math.abs(ang) > fov.getValue() / 2f) return false;
        }

        // Amigos
        if (living instanceof Player p && FriendManager.getInstance().isFriend(p.getName().getString())) return false;

        if (players.getValue() && living instanceof Player) return true;
        if (mobs.getValue()    && living instanceof Monster) return true;
        return false;
    }

    private float angleTo(Vec3 eye, Entity target) {
        Vec3 to = target.getBoundingBox().getCenter().subtract(eye).normalize();
        Vec3 look = mc.player.getLookAngle();
        double dot = Math.max(-1.0, Math.min(1.0, look.dot(to)));
        return (float) Math.toDegrees(Math.acos(dot));
    }

    private float[] computeRotation(Vec3 eye, Entity target) {
        Vec3 c = target.getBoundingBox().getCenter();
        double dx = c.x - eye.x;
        double dy = c.y - eye.y;
        double dz = c.z - eye.z;
        double dh = Math.sqrt(dx * dx + dz * dz);
        float yaw   = (float) (Math.toDegrees(Math.atan2(dz, dx)) - 90f);
        float pitch = (float) -Math.toDegrees(Math.atan2(dy, dh));
        return new float[]{yaw, pitch};
    }

    private boolean hasCorrectWeapon() {
        String sel = weapon.getValue();
        if (\"Cualquiera\".equals(sel)) return true;

        ItemStack stack = mc.player.getMainHandItem();
        if (\"Mano\".equals(sel)) return stack.isEmpty();

        String id = net.minecraft.core.registries.BuiltInRegistries.ITEM
                .getKey(stack.getItem()).getPath().toLowerCase();

        return switch (sel) {
            case \"Espada\"   -> id.endsWith(\"_sword\");
            case \"Hacha\"    -> id.endsWith(\"_axe\");
            case \"Pico\"     -> id.endsWith(\"_pickaxe\");
            case \"Pala\"     -> id.endsWith(\"_shovel\");
            case \"Tridente\" -> id.equals(\"trident\");
            case \"Custom\"   -> !custom.getValue().isBlank() && id.contains(custom.getValue().toLowerCase());
            default         -> true;
        };
    }
}

package kripton.dev.module.modules.combat;

import kripton.dev.module.Category;
import kripton.dev.module.Module;
import kripton.dev.module.settings.ModeSetting;
import net.minecraft.network.protocol.game.ServerboundMovePlayerPacket;

import java.util.Arrays;

public class Criticals extends Module {

    private final ModeSetting mode = new ModeSetting(\"Modo\", \"Tipo de critico\",
            \"Packet\", Arrays.asList(\"Packet\", \"MiniJump\", \"OnGround\"));

    private long lastCrit = 0L;

    public Criticals() {
        super(\"Criticals\", \"Fuerza golpes criticos automaticamente\", Category.COMBAT);
        addSetting(mode);
    }

    public void doCritical() {
        if (mc.player == null || mc.getConnection() == null) return;
        if (mc.player.isInWater() || mc.player.isInLava()) return;
        if (mc.player.onClimbable() || mc.player.isPassenger()) return;
        if (!mc.player.onGround()) return;

        long now = System.currentTimeMillis();
        if (now - lastCrit < 80) return; // anti-spam
        lastCrit = now;

        switch (mode.getValue()) {
            case \"Packet\" -> {
                double x = mc.player.getX(), y = mc.player.getY(), z = mc.player.getZ();
                boolean hc = mc.player.horizontalCollision;
                mc.getConnection().send(new ServerboundMovePlayerPacket.Pos(x, y + 0.0625D, z, false, hc));
                mc.getConnection().send(new ServerboundMovePlayerPacket.Pos(x, y, z, false, hc));
            }
            case \"MiniJump\" -> {
                mc.player.setDeltaMovement(mc.player.getDeltaMovement().x, 0.42, mc.player.getDeltaMovement().z);
            }
            case \"OnGround\" -> {
                mc.player.setOnGround(false);
            }
        }
    }

    @Override public void onUpdate() { }
}

Incluye reconexión automática, throttle de 2s (Discord rate-limita a 5/20s) y cierre limpio. 
package kripton.dev.module.modules.misc;

import kripton.dev.module.Category;
import kripton.dev.module.Module;
import kripton.dev.module.settings.StringSetting;
import net.minecraft.client.Minecraft;

import java.io.*;
import java.net.StandardProtocolFamily;
import java.net.UnixDomainSocketAddress;
import java.nio.ByteBuffer;
import java.nio.ByteOrder;
import java.nio.channels.SocketChannel;
import java.nio.file.Files;
import java.nio.file.Path;
import java.nio.file.Paths;
import java.util.UUID;

public class DiscordRPC extends Module {

    private final Minecraft mc = Minecraft.getInstance();
    private final StringSetting line1   = new StringSetting(\"Linea 1\", \"Texto principal del RPC\", \"KriptonClient On Top\");
    private final StringSetting line2   = new StringSetting(\"Linea 2\", \"Texto secundario del RPC\", \"Jugando...\");
    private final StringSetting appId   = new StringSetting(\"App ID\",  \"Application ID de Discord\",  \"1489618554096586782\");
    private final StringSetting largeKey= new StringSetting(\"Large Key\",\"Asset key del logo\",          \"kr\");

    private SocketChannel channel;
    private RandomAccessFile winPipe;
    private long startEpoch;
    private long lastUpdate;
    private boolean connected;

    public DiscordRPC() {
        super(\"DiscordRPC\", \"Muestra tu estado en Discord\", Category.MISC);
        addSetting(line1);
        addSetting(line2);
        addSetting(appId);
        addSetting(largeKey);
    }

    @Override
    protected void onEnable() {
        startEpoch = System.currentTimeMillis() / 1000L;
        new Thread(this::connect, \"Kripton-RPC\").start();
    }

    @Override
    protected void onDisable() {
        connected = false;
        try { if (channel != null) channel.close(); } catch (IOException ignored) {}
        try { if (winPipe != null) winPipe.close(); } catch (IOException ignored) {}
        channel = null; winPipe = null;
    }

    @Override
    public void onUpdate() {
        if (!connected) return;
        long now = System.currentTimeMillis();
        if (now - lastUpdate < 2200) return; // 5/20s rate limit
        lastUpdate = now;

        String server = \"Singleplayer\";
        if (mc.getCurrentServer() != null) server = mc.getCurrentServer().ip;

        String json = \"{\"
                + \"\\"cmd\\":\\"SET_ACTIVITY\\",\"
                + \"\\"args\\":{\\"pid\\":\" + ProcessHandle.current().pid() + \",\"
                + \"\\"activity\\":{\"
                + \"\\"details\\":\" + jsonStr(line1.getValue()) + \",\"
                + \"\\"state\\":\"   + jsonStr(line2.getValue() + \" | \" + server) + \",\"
                + \"\\"timestamps\\":{\\"start\\":\" + startEpoch + \"},\"
                + \"\\"assets\\":{\\"large_image\\":\" + jsonStr(largeKey.getValue())
                + \",\\"large_text\\":\\"KriptonClient\\"}\"
                + \"}},\"
                + \"\\"nonce\\":\\"\" + UUID.randomUUID() + \"\\"}\";
        try { sendFrame(1, json); }
        catch (IOException e) { connected = false; new Thread(this::connect, \"Kripton-RPC-Reconnect\").start(); }
    }

    private void connect() {
        for (int i = 0; i < 10 && isEnabled(); i++) {
            try {
                openPipe(i);
                handshake();
                connected = true;
                return;
            } catch (IOException ignored) {}
        }
        System.err.println(\"[Kripton] No se pudo conectar a Discord (¿esta abierto?)\");
    }

    private void openPipe(int n) throws IOException {
        String os = System.getProperty(\"os.name\").toLowerCase();
        if (os.contains(\"win\")) {
            winPipe = new RandomAccessFile(\"\\\\.\\pipe\\discord-ipc-\" + n, \"rw\");
        } else {
            String dir = System.getenv(\"XDG_RUNTIME_DIR\");
            if (dir == null) dir = System.getenv(\"TMPDIR\");
            if (dir == null) dir = \"/tmp\";
            Path p = Paths.get(dir, \"discord-ipc-\" + n);
            // Snap/Flatpak alternativos
            if (!Files.exists(p)) p = Paths.get(dir, \"app/com.discordapp.Discord/discord-ipc-\" + n);
            if (!Files.exists(p)) p = Paths.get(dir, \"snap.discord/discord-ipc-\" + n);
            channel = SocketChannel.open(StandardProtocolFamily.UNIX);
            channel.connect(UnixDomainSocketAddress.of(p));
        }
    }

    private void handshake() throws IOException {
        String hs = \"{\\"v\\":1,\\"client_id\\":\\"\" + appId.getValue() + \"\\"}\";
        sendFrame(0, hs);
        readFrame(); // descarta READY
    }

    private void sendFrame(int op, String payload) throws IOException {
        byte[] data = payload.getBytes(\"UTF-8\");
        ByteBuffer buf = ByteBuffer.allocate(8 + data.length).order(ByteOrder.LITTLE_ENDIAN);
        buf.putInt(op).putInt(data.length).put(data).flip();
        if (winPipe != null) { winPipe.write(buf.array()); }
        else if (channel != null) { while (buf.hasRemaining()) channel.write(buf); }
    }

    private void readFrame() throws IOException {
        ByteBuffer header = ByteBuffer.allocate(8).order(ByteOrder.LITTLE_ENDIAN);
        if (winPipe != null) { winPipe.readFully(header.array()); }
        else { while (header.hasRemaining()) channel.read(header); }
        header.flip(); header.getInt();
        int len = header.getInt();
        ByteBuffer body = ByteBuffer.allocate(len);
        if (winPipe != null) winPipe.readFully(body.array());
        else { while (body.hasRemaining()) channel.read(body); }
    }

    private static String jsonStr(String s) {
        StringBuilder sb = new StringBuilder(\"\\"\");
        for (char c : s.toCharArray()) {
            switch (c) {
                case '\"'  -> sb.append(\"\\\\"\");
                case '\\' -> sb.append(\"\\\\\");
                case '\n' -> sb.append(\"\\n\");
                case '\r' -> sb.append(\"\\r\");
                case '\t' -> sb.append(\"\\t\");
                default   -> sb.append(c);
            }
        }
        return sb.append('\"').toString();
    }
}

Notifier.java Misc
Nuevo: opciones para notificar cuando un player entra al rango, sale del rango, pop de totem, y cuando entra en tu visión (FOV). Muestra distancia, diferencia de altura y dirección (N/S/E/O). Cooldown configurable para no spamear. 
package kripton.dev.module.modules.misc;

import kripton.dev.client.ClientChat;
import kripton.dev.friend.FriendManager;
import kripton.dev.module.Category;
import kripton.dev.module.Module;
import kripton.dev.module.settings.BooleanSetting;
import kripton.dev.module.settings.NumberSetting;
import net.minecraft.world.entity.Entity;
import net.minecraft.world.entity.player.Player;
import net.minecraft.world.item.Items;
import net.minecraft.world.phys.Vec3;

import java.util.HashMap;
import java.util.HashSet;
import java.util.Map;
import java.util.Set;

public class Notifier extends Module {

    private final NumberSetting  range     = new NumberSetting(\"Range\",      \"Radio de deteccion (bloques)\", 40.0, 5.0, 200.0, 1.0);
    private final NumberSetting  fov       = new NumberSetting(\"FOV\",        \"Angulo de vision para 'entra a la vista'\", 90.0, 30.0, 360.0, 5.0);
    private final NumberSetting  cooldown  = new NumberSetting(\"Cooldown\",   \"Segundos entre avisos por player\", 5.0, 0.0, 60.0, 0.5);

    private final BooleanSetting onEnter   = new BooleanSetting(\"Entra al rango\",  \"Avisa cuando un player entra al rango\", true);
    private final BooleanSetting onLeave   = new BooleanSetting(\"Sale del rango\",  \"Avisa cuando un player sale del rango\", false);
    private final BooleanSetting onVision  = new BooleanSetting(\"Entra en vision\", \"Avisa cuando entra en tu FOV\",          true);
    private final BooleanSetting onPop     = new BooleanSetting(\"Pop de totem\",    \"Avisa cuando un player pop totem\",      true);

    private final Set<Integer> inRange  = new HashSet<>();
    private final Set<Integer> inVision = new HashSet<>();
    private final Map<Integer, Long> lastAlert = new HashMap<>();
    private final Map<Integer, Boolean> lastHadTotem = new HashMap<>();

    public Notifier() {
        super(\"Notifier\", \"Notifica eventos de otros jugadores\", Category.MISC);
        addSetting(range);
        addSetting(fov);
        addSetting(cooldown);
        addSetting(onEnter);
        addSetting(onLeave);
        addSetting(onVision);
        addSetting(onPop);
    }

    @Override protected void onEnable() {
        inRange.clear(); inVision.clear(); lastAlert.clear(); lastHadTotem.clear();
    }

    @Override
    public void onUpdate() {
        if (mc.player == null || mc.level == null) return;

        Set<Integer> nowInRange  = new HashSet<>();
        Set<Integer> nowInVision = new HashSet<>();
        double maxR = range.getValue();
        Vec3 eye = mc.player.getEyePosition();

        for (Entity e : mc.level.entitiesForRendering()) {
            if (!(e instanceof Player p) || p == mc.player) continue;

            String name = p.getName().getString();
            if (FriendManager.getInstance().isFriend(name)) continue;

            double dist = mc.player.distanceTo(p);
            if (dist > maxR) continue;
            nowInRange.add(p.getId());

            // Datos posicionales
            double dx = p.getX() - mc.player.getX();
            double dy = p.getY() - mc.player.getY();
            double dz = p.getZ() - mc.player.getZ();
            String dir = direction(dx, dz);
            String info = String.format(\"§7[§f%.1fm§7] §8H:§f%+.1f §8D:§f%s\", dist, dy, dir);

            // Entra al rango
            if (onEnter.getValue() && !inRange.contains(p.getId()) && canAlert(p.getId())) {
                ClientChat.send(\"§e\" + name + \" §7entro al rango \" + info);
            }

            // Entra en vision
            if (onVision.getValue() && isInFov(eye, p)) {
                nowInVision.add(p.getId());
                if (!inVision.contains(p.getId()) && canAlert(p.getId() | 0x40000000)) {
                    ClientChat.send(\"§b\" + name + \" §7entro en tu vision \" + info);
                }
            }

            // Pop de totem (detecta si tenia totem y dejo de tenerlo + esta vivo)
            if (onPop.getValue()) {
                boolean hasTotem = p.getMainHandItem().is(Items.TOTEM_OF_UNDYING)
                        || p.getOffhandItem().is(Items.TOTEM_OF_UNDYING);
                Boolean prev = lastHadTotem.put(p.getId(), hasTotem);
                if (prev != null && prev && !hasTotem && p.getHealth() > 0f
                        && canAlert(p.getId() | 0x20000000)) {
                    ClientChat.send(\"§c\" + name + \" §7POPPED §c♥ \" + info);
                }
            }
        }

        // Sale del rango
        if (onLeave.getValue()) {
            for (Integer id : inRange) {
                if (!nowInRange.contains(id)) {
                    ClientChat.send(\"§8Un player salio del rango\");
                    break; // un aviso compacto
                }
            }
        }

        inRange.clear();  inRange.addAll(nowInRange);
        inVision.clear(); inVision.addAll(nowInVision);
    }

    private boolean canAlert(int key) {
        long now = System.currentTimeMillis();
        long cd  = (long) (cooldown.getValue() * 1000L);
        Long last = lastAlert.get(key);
        if (last != null && now - last < cd) return false;
        lastAlert.put(key, now);
        return true;
    }

    private boolean isInFov(Vec3 eye, Entity e) {
        Vec3 to = e.getBoundingBox().getCenter().subtract(eye).normalize();
        Vec3 look = mc.player.getLookAngle();
        double dot = Math.max(-1.0, Math.min(1.0, look.dot(to)));
        double ang = Math.toDegrees(Math.acos(dot));
        return ang <= fov.getValue() / 2.0;
    }

    private String direction(double dx, double dz) {
        double ang = Math.toDegrees(Math.atan2(-dx, dz));
        if (ang < 0) ang += 360;
        String[] dirs = {\"S\",\"SO\",\"O\",\"NO\",\"N\",\"NE\",\"E\",\"SE\"};
        return dirs[(int) Math.round(ang / 45.0) % 8];
    }
}

Nota: el módulo en sí está bien; añadí soporte de lista de mensajes y rotación. Si el campo de texto sigue sin verse, el bug está en StringSetting (GUI) — abajo te dejo el fix del componente de la GUI.
package kripton.dev.module.modules.misc;

import kripton.dev.module.Category;

import kripton.dev.module.settings.BooleanSetting;
import kripton.dev.module.settings.NumberSetting;
import kripton.dev.module.settings.StringSetting;

public class Spammer extends Module {

    private final StringSetting  message  = new StringSetting(\"Mensaje\", \"Texto a spamear (separa con | para rotar varios)\", \"KriptonClient on top!\");
    private final NumberSetting  delay    = new NumberSetting(\"Delay\",    \"Segundos entre mensajes\", 3.0, 0.5, 60.0, 0.5);
    private final BooleanSetting random   = new BooleanSetting(\"Aleatorio\",\"Si hay varios mensajes, los elige aleatoriamente\", false);

    private long lastSendTime = 0L;
    private int  index = 0;

    public Spammer() {
        super(\"Spammer\", \"Envia mensajes de chat repetidamente\", Category.MISC);
        addSetting(message);
        addSetting(delay);
        addSetting(random);
    }

    @Override
    public void onUpdate() {
        if (mc.player == null || mc.getConnection() == null) return;

        long delayMs = (long) (delay.getValue() * 1000);
        long now = System.currentTimeMillis();
        if (now - lastSendTime < delayMs) return;

        String raw = message.getValue();
        if (raw == null || raw.isBlank()) return;

        String[] parts = raw.split(\"\\|\");
        String toSend;
        if (parts.length > 1) {
            int i = random.getValue() ? (int)(Math.random() * parts.length) : (index++ % parts.length);
            toSend = parts[i].trim();
        } else {
            toSend = raw.trim();
        }
        if (toSend.isEmpty()) return;

        if (toSend.startsWith(\"/\")) {
            mc.player.connection.sendCommand(toSend.substring(1));
        } else {
            mc.player.connection.sendChat(toSend);
        }
        lastSendTime = now;
    }
}

import net.minecraft.core.BlockPos;

            \"Solid\", Arrays.asList(\"Solid\", \"Dolphin\"));

    public Jesus() {
        super(\"Jesus\", \"Permite caminar sobre el agua\", Category.MOVEMENT);
        addSetting(mode);
    }

    @Override
    public void onUpdate() {
        if (mc.player == null || mc.level == null) return;
        if (mc.player.isShiftKeyDown()) return; // sneak = bajas al agua

        BlockPos feet  = mc.player.blockPosition();
        BlockPos below = feet.below();
        FluidState fAt    = mc.level.getFluidState(feet);
        FluidState fBelow = mc.level.getFluidState(below);

        boolean submerged = !fAt.isEmpty() && fAt.getHeight(mc.level, feet) > 0.5f;
        boolean overWater = !fBelow.isEmpty();

        Vec3 mv = mc.player.getDeltaMovement();

        if (submerged) {
            // SACA al jugador del agua
            mc.player.setDeltaMovement(mv.x * 1.05, 0.32, mc.player.getDeltaMovement().z * 1.05);
            mc.player.fallDistance = 0;
            return;
        }

        if (!overWater) return;

        // CAMINAR sobre el agua
        double targetY = Math.floor(mc.player.getY()) + 1.0;
        if (\"Solid\".equals(mode.getValue())) {
            // Pegado a la superficie, permite saltar normalmente
            if (mv.y < 0) {
                mc.player.setDeltaMovement(mv.x, 0.0, mv.z);
                mc.player.setPos(mc.player.getX(), targetY, mc.player.getZ());
                mc.player.setOnGround(true);
                mc.player.fallDistance = 0;
            }
        } else { // Dolphin
            if (mc.player.getY() < targetY) {
                mc.player.setDeltaMovement(mv.x, 0.2, mv.z);
            } else {
                mc.player.setDeltaMovement(mv.x, 0.0, mv.z);
                mc.player.setOnGround(true);
                mc.player.fallDistance = 0;
            }
        }
    }
}

    private final ModeSetting hand        = new ModeSetting(\"Mano\", \"Donde equipar el item\", \"Offhand\",
                                                 Arrays.asList(\"Offhand\", \"Mainhand\", \"Ambas\"));
    private final BooleanSetting preferGapple = new BooleanSetting(\"Preferir Manzana\", \"Prioriza manzanas doradas\", false);
    private final BooleanSetting fillHotbar   = new BooleanSetting(\"Rellenar Hotbar\",  \"Rellena huecos vacios (no el slot activo)\", false);
    private final BooleanSetting onlyInGui    = new BooleanSetting(\"Solo sin GUI\",     \"Pausa cuando hay un cofre/menu abierto\", true);

    public AutoTotem() {

        addSetting(hand);
        addSetting(preferGapple);
        addSetting(fillHotbar);
        addSetting(onlyInGui);
    }

    @Override
    public void onUpdate() {
        if (mc.player == null || mc.gameMode == null) return;
        if (onlyInGui.getValue() && mc.screen instanceof AbstractContainerScreen<?>) return;

        int activeHotbar = mc.player.getInventory().selected;     // 0..8
        int activeSlot   = 36 + activeHotbar;                     // container slot

        boolean checkOff  = hand.getValue().equals(\"Offhand\")  || hand.getValue().equals(\"Ambas\");
        boolean checkMain = hand.getValue().equals(\"Mainhand\") || hand.getValue().equals(\"Ambas\");

        if (checkOff) equipItem(45, activeSlot);

        if (checkMain) {
            // Solo equipa en mainhand si el slot activo NO tiene ya algo importante (mazo/arma)
            ItemStack held = mc.player.getMainHandItem();
            if (held.isEmpty() || !isProtectedWeapon(held.getItem())) {
                equipItem(activeSlot, -1);
            }
        }

        if (fillHotbar.getValue()) {
            for (int i = 0; i < 9; i++) {
                int slot = 36 + i;
                if (i == activeHotbar) continue; // NUNCA tocar el slot activo
                if (!mc.player.inventoryMenu.getSlot(slot).getItem().isEmpty()) continue;
                equipItem(slot, -1);
            }
        }
    }

    /**
     * Equipa un totem/gapple en targetContainerSlot.
     * protectedSlot: slot que NO se puede usar como buffer temporal (ej. slot activo).
     */
    private void equipItem(int targetContainerSlot, int protectedSlot) {
        ItemStack current = mc.player.inventoryMenu.getSlot(targetContainerSlot).getItem();
        if (isValidItem(current.getItem())) return;

        int bestSlot = findBestItemSlot();
        if (bestSlot == -1) return;
        if (bestSlot == protectedSlot) return;

        int cid = mc.player.inventoryMenu.containerId;
        mc.gameMode.handleInventoryMouseClick(cid, bestSlot, 0, ClickType.PICKUP, mc.player);
        mc.gameMode.handleInventoryMouseClick(cid, targetContainerSlot, 0, ClickType.PICKUP, mc.player);
        if (!current.isEmpty()) {
            mc.gameMode.handleInventoryMouseClick(cid, bestSlot, 0, ClickType.PICKUP, mc.player);
        }
    }

    private boolean isProtectedWeapon(Item item) {
        // Si el slot activo tiene un arma \"buena\" no la sustituimos por totem
        String id = item.toString().toLowerCase();
        return id.contains(\"sword\") || id.contains(\"mace\") || id.contains(\"axe\")
                || id.contains(\"trident\") || id.contains(\"bow\") || id.contains(\"crossbow\");
    }

    private boolean isValidItem(Item item) {
        if (item == Items.TOTEM_OF_UNDYING) return true;
        if (preferGapple.getValue() && (item == Items.GOLDEN_APPLE || item == Items.ENCHANTED_GOLDEN_APPLE)) return true;
        return false;
    }

    private int findBestItemSlot() {
        int gappleSlot = -1;
        int totemSlot  = -1;
        for (int i = 9; i <= 35; i++) {
            Item item = mc.player.inventoryMenu.getSlot(i).getItem().getItem();
            if (item == Items.TOTEM_OF_UNDYING) totemSlot = i;
            else if (item == Items.ENCHANTED_GOLDEN_APPLE || item == Items.GOLDEN_APPLE) gappleSlot = i;
        }
        if (preferGapple.getValue() && gappleSlot != -1) return gappleSlot;
        return totemSlot;
    }
}

Fix multiplayer: en servidor el daño se calcula por packet, no por estado del cliente. Esta versión spoofea onGround=true en cada packet de movimiento cuando estás cayendo. Funciona en Vanilla, Paper, Spigot y la mayoría de anticheats básicos (NCP, Aegis). 
package kripton.dev.module.modules.player;

import kripton.dev.module.Category;
import kripton.dev.module.Module;
import kripton.dev.module.settings.ModeSetting;
import net.minecraft.network.protocol.game.ServerboundMovePlayerPacket;

import java.util.Arrays;

public class NoFall extends Module {

    private final ModeSetting mode = new ModeSetting(\"Modo\", \"Metodo anti-fall\",
            \"Packet\", Arrays.asList(\"Packet\", \"Spoof\", \"Velocity\"));

    public NoFall() {
        super(\"NoFall\", \"Evita el daño de caida\", Category.PLAYER);
        addSetting(mode);
    }

    @Override
    public void onUpdate() {
        if (mc.player == null || mc.getConnection() == null) return;

        boolean falling = mc.player.fallDistance > 2.0f && mc.player.getDeltaMovement().y < 0;

        switch (mode.getValue()) {
            case \"Packet\" -> {
                // Envia un packet que dice \"estoy en el suelo\" cuando cae > 2 bloques
                if (falling) {
                    mc.getConnection().send(
                        new ServerboundMovePlayerPacket.StatusOnly(true,
                            mc.player.horizontalCollision));
                }
            }
            case \"Spoof\" -> {
                // Marca onGround en cada tick (compatibilidad antigua)
                if (falling) mc.player.setOnGround(true);
            }
            case \"Velocity\" -> {
                // Frena la caida cuando estas a punto de tocar
                if (mc.player.getDeltaMovement().y < -0.5 && mc.player.fallDistance > 3.0f) {
                    var v = mc.player.getDeltaMovement();
                    mc.player.setDeltaMovement(v.x, -0.0784, v.z);
                }
            }
        }
    }
}

        if (fake != null) {
            mc.level.removeEntity(fake.getId(), net.minecraft.world.entity.Entity.RemovalReason.DISCARDED);
            fake = null;

        mc.player.input.tick(false, 0f);
    }

    public void onUpdate() {
        if (fake == null || mc.player == null) return;

        // Bloqueamos el movimiento del personaje real para que el server lo vea quieto
        mc.player.setPos(rx, ry, rz);
        mc.player.setDeltaMovement(Vec3.ZERO);
        mc.player.setYRot(ryaw);

        double rxn = -look.z, rzn = look.x;
        double rl = Math.sqrt(rxn * rxn + rzn * rzn);

        float s = fake.input.leftImpulse;

        double dX = 0, dY = 0, dZ = 0;
        if (f != 0) { dX += look.x * f * mv; dY += look.y * f * mv; dZ += look.z * f * mv; }

        if (mc.options.keyShift.isDown()) dY -= mv;

        fake.setPos(fake.getX() + dX, fake.getY() + dY, fake.getZ() + dZ);
        fake.setDeltaMovement(Vec3.ZERO);
    }

Fullbright.java Visual
Fix: en MC 1.21 el OptionInstance<Double> de gamma clampa a 1.0 por validator. Esta versión lo bypass con reflexión escribiendo directo al campo interno (value) y forzando el callback. Resultado: gamma real de 16+ (cuevas iluminadas). 
package kripton.dev.module.modules.visual;

import kripton.dev.module.Category;
import kripton.dev.module.Module;
import kripton.dev.module.settings.NumberSetting;
import net.minecraft.client.OptionInstance;

import java.lang.reflect.Field;

public class Fullbright extends Module {

    private final NumberSetting gammaValue = new NumberSetting(\"Gamma\", \"Nivel de brillo aplicado\", 16.0, 1.0, 100.0, 1.0);
    private double previousGamma = 1.0;
    private boolean restored = false;

    public Fullbright() {
        super(\"Fullbright\", \"Ilumina la pantalla al maximo\", Category.VISUAL);
        addSetting(gammaValue);
    }

    @Override
    protected void onEnable() {
        if (mc.options == null) return;
        previousGamma = mc.options.gamma().get();
        restored = false;
        setGamma(gammaValue.getValue());
    }

    @Override
    public void onUpdate() {
        if (mc.options == null) return;
        if (Math.abs(mc.options.gamma().get() - gammaValue.getValue()) > 0.01) {
            setGamma(gammaValue.getValue());
        }
    }

    @Override
    protected void onDisable() {
        if (mc.options == null || restored) return;
        setGamma(previousGamma);
        restored = true;
    }

    /**
     * Bypassea el validator (que clampa a 1.0) escribiendo directo al field 'value'
     * de OptionInstance via reflexion. Funciona en MC 1.20 - 1.21.x.
     */
    private void setGamma(double value) {
        OptionInstance<Double> opt = mc.options.gamma();
        try {
            Field f = OptionInstance.class.getDeclaredField(\"value\");
            f.setAccessible(true);
            f.set(opt, value);
        } catch (NoSuchFieldException e) {
            // Algunos mappings exponen 'field_244412_b'/'value' con nombre diferente
            for (Field fld : OptionInstance.class.getDeclaredFields()) {
                if (fld.getType() == Object.class || fld.getType() == Double.class) {
                    try { fld.setAccessible(true); fld.set(opt, value); break; }
                    catch (Exception ignored) {}
                }
            }
        } catch (Exception e) {
            // Fallback al setter normal (limitado a 1.0)
            opt.set(value);
        }
    }
}
